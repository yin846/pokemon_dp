NNS_G2dSetAnimCtrlCurrentTime

C Specification

#include <nnsys/g2d/g2d_Animation.h>

    ( 
        NNSG2dAnimController*       pAnimCtrl,
        fx32                        time
    );

Arguments

pAnimCtrl [OUT] NNSG2dAnimController entity
