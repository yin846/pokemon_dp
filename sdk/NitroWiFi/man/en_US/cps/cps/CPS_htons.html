CPS_htons
CPS_htonl

C Specification

#include <nitroWiFi/cps.h>
static inline u16
CPS_htons (u16 data)
{
return (u16)((data >> 8) | (data << 8));
}

static inline u32
CPS_htonl(u32 data)
{
u32 tmp;

tmp = ((data >> 8) & 0x00ff00ff) | ((data << 8) & 0xff00ff00);
return (tmp >> 16) | (tmp << 16);
}

Arguments

data Host byte order value.

Return Values

Network byte order value.

Description

Converts the host byte order value to a network byte order value.
